### 项目单独指定maven远程仓库

`pom.xml`

```
<!-- 配置maven项目的远程仓库 -->
<repositories>
    <repository>
        <id>aliyun-repos</id>
        <name>aliyun-repos</name>
        <url>https://maven.aliyun.com/nexus/content/groups/public/</url>
        <!-- 是否开启发布版构件下载 -->
        <releases>
            <enabled>true</enabled>
        </releases>
        <!-- 是否开启快照版构件下载 -->
        <snapshots>
            <enabled>false</enabled>
        </snapshots>
    </repository>
</repositories>

<!-- 配置maven插件的远程仓库 -->
<pluginRepositories>
    <pluginRepository>
        <id>aliyun-plugin</id>
        <name>aliyun-plugin</name>
        <url>https://maven.aliyun.com/nexus/content/groups/public/</url>
        <releases>
            <enabled>true</enabled>
        </releases>
        <snapshots>
            <enabled>false</enabled>
        </snapshots>
    </pluginRepository>
</pluginRepositories>
```
