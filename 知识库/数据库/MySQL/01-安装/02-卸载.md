# 卸载

```shell
# ps -ef | grep mysql

# 查看mysql安装
rpm -qa|grep -i mysql
# 卸载
rpm -ev bt-mysql57-5.7.37-1.el7.x86_64


# 卸载mariadb数据库（MySQL的一个分支）
rpm -qa |grep mariadb
rpm -ev  mariadb-libs-5.5.68-1.el7.x86_64


# 查找之前老版本mysql的目录、并且删除老版本mysql的文件和库
find / -name mysql 

rm -rf /var/lib/mysql
rm -rf /home/mysql

rm -rf /etc/my.cnf


# 删除mysql用户和组
id mysql
userdel mysql
```
